003  (defn capture-group-names
004    "If you have a regex with named capture groups like:
005     `#\"(?<year>\\d+)\\-(?<month>\\d+)\\-(?<day>\\d+)\"`
006     This function will return a list like: `[\"year\", \"month\", \"day\"]`"
007    [regex]
008    (let [matcher (re-matcher #"\<(\w+)\>" (str regex))]
009      (loop [result []]
010        (let [[_ group-name :as match] (re-find matcher)]
011          (if-not (seq match)
012            result
013            (recur (conj result group-name)))))))
014  
015  (defn named-capture-groups
016    "If your regex has named capture groups
017     This function will return a map of `<group-name>:<matched-string>`"
018    [regex input]
019    (let [matcher (re-matcher regex input)]
020      (when (.matches matcher)
021        (->> (capture-group-names regex)
022             (map (fn [n] [(keyword n) (.group matcher n)]))
023             (into {})))))
